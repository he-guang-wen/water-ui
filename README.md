


# Install

```javascript
# Using npm
npm i water-next -S
```

# Quickstart
```javascript
import { createApp } from 'vue';
import App from './App.vue'
import waterUi from  'water-next'
import 'water-next/components/style/scss/index.scss'
const app = createApp(App)
app.use(waterUi)
app.mount('#app')
```
